```php
public function perform(): void
{
    $config = ConfigFactory::fromKirbyOptions();
    
    if (!$config->isConfigured()) {
        throw new VerificationException(
            VerificationException::NOT_CONFIGURED,
            'Mosparo is not configured'
        );
    }
    
    // Extract tokens
    $submitToken = $this->data['_mosparo_submitToken'] ?? null;
    $validationToken = $this->data['_mosparo_validationToken'] ?? null;
    
    if (empty($submitToken) || empty($validationToken)) {
        $this->reject('mosparo.error.tokens_missing', 'mosparo');
        return;
    }
    
    // Create service and verify
    $service = new VerificationService($config);
    
    try {
        // Prepare form data (removes tokens and ignored fields)
        $formData = $service->prepareFormData($this->data);
        
        // Verify submission
        $result = $service->verify($formData, $submitToken, $validationToken);
        
        // Check if submittable
        if (!$result->isSubmittable()) {
            $this->reject('mosparo.error.verification_failed', 'mosparo');
            return;
        }
        
        // BYPASS PROTECTION: Verify required fields were actually checked
        // Get fields that should have been verified (all non-ignored form fields)
        $expectedFields = array_keys($formData);
        
        if (!$service->verifyRequiredFields($result, $expectedFields)) {
            // Log suspicious activity (without exposing details)
            error_log('Mosparo: Potential bypass attempt detected');
            $this->reject('mosparo.error.bypass_detected', 'mosparo');
            return;
        }
        
    } catch (\Mosparo\ApiClient\Exception $e) {
        // Log sanitized error (no private key in message)
        error_log('Mosparo API error: ' . $this->sanitizeErrorMessage($e->getMessage()));
        $this->reject('mosparo.error.api_error', 'mosparo');
        return;
    }
    
    // Success - guard passed
}
```

3. **Add private helper method:**
```php
/**
 * Sanitize error message to remove sensitive data
 */
private function sanitizeErrorMessage(string $message): string
{
    // Remove any URLs that might contain keys
    $message = preg_replace('/https?:\/\/[^\s]+/', '[URL_REDACTED]', $message);
    // Remove potential key patterns
    $message = preg_replace('/[a-f0-9]{32,}/i', '[KEY_REDACTED]', $message);
    return $message;
}
```
